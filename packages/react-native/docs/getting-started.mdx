# Getting Started

Wrap your application with the `HearthUIProvider` component.

```tsx
import React from 'react';
import { HearthUIProvider } from '@utilitywarehouse/hearth-react-native';
import App from './App';

const RootApp: React.FC = () => (
  <HearthUIProvider>
    <App />
  </HearthUIProviderr>
);
```

## Dark Mode

By default the light colour mode is enabled. If you want to enable dark mode, you can pass the `colorMode` prop to the `HearthUIProvider` component. The two options for `colorMode` are `light` and `dark`.

```tsx
import React from 'react';
import { ColorSchemeName, useColorScheme } from 'react-native';
import {
  StyleSheet,
  breakpoints,
  themes,
  UnistylesRuntime,
} from '@utilitywarehouse/hearth-react-native';
import App from './App';

StyleSheet.configure({
  breakpoints,
  themes,
  settings: {
    initialTheme: 'light',
    adaptiveThemes: false,
  },
});

const RootApp: React.FC = () => {
  const colorScheme = useColorScheme();

  useEffect(() => {
    UnistylesRuntime.setTheme(colorScheme as ColorSchemeName);
  }, [colorScheme]);

  return <App />;
};
```
